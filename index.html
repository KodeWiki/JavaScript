NaN

NaN is number which represents a non-sensical mathematical statement. To check if a variable is NaN, isNaN() method should be used

typeof NaN                  // 'number'

isNaN(3)                    // false
isNaN('3')                  // false
isNaN('three')              // true

console.log(NaN == NaN)     // one non-sensical math op is not equal to other
(9 / 0) === (8 / 0)         // true! how can one infinite number be equal to another?

Type Coercion

Automatic type coercion - When an operator is applied to wrong type of values then js quietly tries converting them to suitable type

console.log('5' - 1)                // 4
console.log('5' + 1)                // 51
console.log('8' * null)             // 0
console.log('five' - 2)             // NaN, "five" couldn't convert to number
console.log(1 == '1')               // true

console.log(null == undefined)      // true, exception : null and undefined
                                    // have same values!

console.log(false == null)          // false, useful to check if some variable
                                    // have real value, compare them to null
                                    // or (undefined).
console.log(null == 0)              // false, real value vs. null
console.log(0 == '')                // true
console.log(0 == '0')               // true

For cases where automatic type coercion is not desirable use === or !==

console.log(0 === '0')              // false
console.log('abc' === 'abc')        // true

Explicit Type Coercion 

console.log(Number('34'))           // 34
console.log(String(34))             // '34'
console.log(Boolean(undefined))     // false
console.log(Number('three'))        // NaN

Booleans

  • Boolean value true, false
  • Boolean Operators &&, ||, ==, !, !=, <=, >=
  • && and || operators : returns last value required for computation of expression. Also in case of these operators an expression is evaluated if necessary
console.log(null || 'b')            // 'b', useful for setting a fallback value
                                    // incase runtime value is empty or null
console.log('a' || 'b')             // 'a'
console.log('a' && 'b')             // 'b'
console.log(null && 'b')            // null
function isTrue(val) {
    boolVal = val ? true : false    // alternative : Boolean(val)
    console.log(val + ' is ' + boolVal)
}

isTrue(NaN)            // false
isTrue(null)           // false
isTrue(undefined)      // false
isTrue('')             // false
isTrue('a')            // true
isTrue(0)              // false
isTrue(1)              // true
isTrue([])             // true
isTrue({})             // true

Conditionals and Loops

if-else statement
var num = 1
if (num < 10)
    console.log('small')
else if (num < 100)
    console.log('medium')
else
    console.log('large')
switch-case statement
var one = 1
switch(one) {
    case '1':
        console.log('one')
        break
    case '2':
        console.log('two')
        break
    default:
        console.log('no suitable match found')
}
